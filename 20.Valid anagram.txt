class Solution {
    public boolean isAnagram(String s, String t)
    {
        char[] ch1=s.toCharArray();
        char[] ch2=t.toCharArray();
        Arrays.sort(ch1);
        Arrays.sort(ch2);
        s=String.valueOf(ch1); 
        t=String.valueOf(ch2); 
        if(s.equals(t))
            return true;
        else
            return false;
    }
}
